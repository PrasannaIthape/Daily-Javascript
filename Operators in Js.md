## Solutions (Javascript)
```

let a=10;
let b=5;

console.log("a: ",a," b: ",b);
console.log("a + b = ",a+b); //15
console.log("a - b = ",a-b); //5
console.log("a * b = ",a*b); //50
console.log("a / b = ",a/b); //2
console.log("a % b = ",a%b); //0  Modulus Operator
console.log("a ** b = ",a**b); // 10^5 
```
